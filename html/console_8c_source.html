8 {

16

18  cons.cur_x = 8;
19  cons.cur_y = 28;
20  cons.cur_c = -1;
21  task->cons = &cons;
22  task->cmdline = cmdline;
23 
24  if (cons.sht != 0) {
25  cons.timer = timer_alloc();
26  timer_init(cons.timer, &task->fifo, 1);
27  timer_settime(cons.timer, 50);
28  }
29  file_readfat(fat, (unsigned char *) (ADR_DISKIMG + 0x000200));
30  for (i = 0; i < 8; i++) {
31  fhandle[i].buf = 0; /* 未使用マーク */
32  }
33  task->fhandle = fhandle;
34  task->fat = fat;
35  if (nihongo[4096] != 0xff) { /* 日本語フォントファイルを読み込めたか？ */
36  task->langmode = 1;
37  } else {
38  task->langmode = 0;
39  }
40  task->langbyte1 = 0;
41 
42  /* プロンプト表示 */
43  cons_putchar(&cons, '>', 1);
44 
45  for (;;) {
46  io_cli();
47  if (fifo32_status(&task->fifo) == 0) {
48  task_sleep(task);
49  io_sti();
50  } else {
51  i = fifo32_get(&task->fifo);
52  io_sti();
53  if (i <= 1 && cons.sht != 0) { /* カーソル用タイマ */
54  if (i != 0) {
55  timer_init(cons.timer, &task->fifo, 0); /* 次は0を */
56  if (cons.cur_c >= 0) {
57  cons.cur_c = COL8_FFFFFF;
58  }
59  } else {
60  timer_init(cons.timer, &task->fifo, 1); /* 次は1を */
61  if (cons.cur_c >= 0) {
62  cons.cur_c = COL8_000000;
63  }
64  }
65  timer_settime(cons.timer, 50);
66  }
67  if (i == 2) { /* カーソルON */
68  cons.cur_c = COL8_FFFFFF;
69  }
70  if (i == 3) { /* カーソルOFF */
71  if (cons.sht != 0) {
72  boxfill8(cons.sht->buf, cons.sht->bxsize, COL8_000000,
73  cons.cur_x, cons.cur_y, cons.cur_x + 7, cons.cur_y + 15);
74  }
75  cons.cur_c = -1;
76  }
77  if (i == 4) { /* コンソールの「×」ボタンクリック */
78  cmd_exit(&cons, fat);
79  }
80  if (256 <= i && i <= 511) { /* キーボードデータ（タスクA経由） */
81  if (i == 8 + 256) {
82  /* バックスペース */
83  if (cons.cur_x > 16) {
84  /* カーソルをスペースで消してから、カーソルを1つ戻す */
85  cons_putchar(&cons, ' ', 0);
86  cons.cur_x -= 8;
87  }
88  } else if (i == 10 + 256) {
89  /* Enter */
90  /* カーソルをスペースで消してから改行する */
91  cons_putchar(&cons, ' ', 0);
92  cmdline[cons.cur_x / 8 - 2] = 0;
93  cons_newline(&cons);
94  cons_runcmd(cmdline, &cons, fat, memtotal); /* コマンド実行 */
95  if (cons.sht == 0) {
96  cmd_exit(&cons, fat);
97  }
98  /* プロンプト表示 */
99  cons_putchar(&cons, '>', 1);
100  } else {
101  /* 一般文字 */
102  if (cons.cur_x < 240) {
103  /* 一文字表示してから、カーソルを1つ進める */
104  cmdline[cons.cur_x / 8 - 2] = i - 256;
105  cons_putchar(&cons, i - 256, 1);
106  }
107  }
108  }
109  /* カーソル再表示 */
110  if (cons.sht != 0) {
111  if (cons.cur_c >= 0) {
112  boxfill8(cons.sht->buf, cons.sht->bxsize, cons.cur_c,
113  cons.cur_x, cons.cur_y, cons.cur_x + 7, cons.cur_y + 15);
114  }
115  sheet_refresh(cons.sht, cons.cur_x, cons.cur_y, cons.cur_x + 8, cons.cur_y + 16);
116  }
117  }
118  }

345 int cmd_app(struct CONSOLE *cons, int *fat, char *cmdline)
346 {
347  struct MEMMAN *memman = (struct MEMMAN *) MEMMAN_ADDR;
348  struct FILEINFO *finfo;
349  char name[18], *p, *q;
350  struct TASK *task = task_now();
351  int i, segsiz, datsiz, esp, dathrb, appsiz;
352  struct SHTCTL *shtctl;
353  struct SHEET *sht;
354 
355  /* コマンドラインからファイル名を生成 */
356  for (i = 0; i < 13; i++) {
357  if (cmdline[i] <= ' ') {
358  break;
359  }
360  name[i] = cmdline[i];
361  }
362  name[i] = 0; /* とりあえずファイル名の後ろを0にする */
363 
364  /* ファイルを探す */
365  finfo = file_search(name, (struct FILEINFO *) (ADR_DISKIMG + 0x002600), 224);
366  if (finfo == 0 && name[i - 1] != '.') {
367  /* 見つからなかったので後ろに".HRB"をつけてもう一度探してみる */
368  name[i ] = '.';
369  name[i + 1] = 'H';
370  name[i + 2] = 'R';
371  name[i + 3] = 'B';
372  name[i + 4] = 0;
373  finfo = file_search(name, (struct FILEINFO *) (ADR_DISKIMG + 0x002600), 224);
374  }
375 
376  if (finfo != 0) {
377  /* ファイルが見つかった場合 */
378  appsiz = finfo->size;
379  p = file_loadfile2(finfo->clustno, &appsiz, fat);
380  if (appsiz >= 36 && strncmp(p + 4, "Hari", 4) == 0 && *p == 0x00) {
381  segsiz = *((int *) (p + 0x0000));
382  esp = *((int *) (p + 0x000c));
383  datsiz = *((int *) (p + 0x0010));
384  dathrb = *((int *) (p + 0x0014));
385  q = (char *) memman_alloc_4k(memman, segsiz);
386  task->ds_base = (int) q;
387  set_segmdesc(task->ldt + 0, appsiz - 1, (int) p, AR_CODE32_ER + 0x60);
388  set_segmdesc(task->ldt + 1, segsiz - 1, (int) q, AR_DATA32_RW + 0x60);
389  for (i = 0; i < datsiz; i++) {
390  q[esp + i] = p[dathrb + i];
391  }
392  start_app(0x1b, 0 * 8 + 4, esp, 1 * 8 + 4, &(task->tss.esp0));
393  shtctl = (struct SHTCTL *) *((int *) 0x0fe4);
394  for (i = 0; i < MAX_SHEETS; i++) {
395  sht = &(shtctl->sheets0[i]);
396  if ((sht->flags & 0x11) == 0x11 && sht->task == task) {
397  /* アプリが開きっぱなしにした下じきを発見 */
398  sheet_free(sht); /* 閉じる */
399  }
400  }
401  for (i = 0; i < 8; i++) { /* クローズしてないファイルをクローズ */
402  if (task->fhandle[i].buf != 0) {
403  memman_free_4k(memman, (int) task->fhandle[i].buf, task->fhandle[i].size);
404  task->fhandle[i].buf = 0;
405  }
406  }
407  timer_cancelall(&task->fifo);
408  memman_free_4k(memman, (int) q, segsiz);
409  task->langbyte1 = 0;
410  } else {
411  cons_putstr0(cons, ".hrb file format error.\n");
412  }
413  memman_free_4k(memman, (int) p, appsiz);
414  cons_newline(cons);
415  return 1;
416  }
417  /* ファイルが見つからなかった場合 */
418  return 0;
419 }

421 int *hrb_api(int edi, int esi, int ebp, int esp, int ebx, int edx, int ecx, int eax)

427  struct SHEET *sht;
428  struct FIFO32 *sys_fifo = (struct FIFO32 *) *((int *) 0x0fec);
429  int *reg = &eax + 1; /* eaxの次の番地 */
430  /* 保存のためのPUSHADを強引に書き換える */
431  /* reg[0] : EDI, reg[1] : ESI, reg[2] : EBP, reg[3] : ESP */
432  /* reg[4] : EBX, reg[5] : EDX, reg[6] : ECX, reg[7] : EAX */
433  int i;

435  struct FILEHANDLE *fh;
436  struct MEMMAN *memman = (struct MEMMAN *) MEMMAN_ADDR;
437 
438  if (edx == 1) {
439  cons_putchar(cons, eax & 0xff, 1);
440  } else if (edx == 2) {
441  cons_putstr0(cons, (char *) ebx + ds_base);
442  } else if (edx == 3) {
443  cons_putstr1(cons, (char *) ebx + ds_base, ecx);
444  } else if (edx == 4) {
445  return &(task->tss.esp0);
446  } else if (edx == 5) {
447  sht = sheet_alloc(shtctl);
448  sht->task = task;
449  sht->flags |= 0x10;
450  sheet_setbuf(sht, (char *) ebx + ds_base, esi, edi, eax);

453  sheet_updown(sht, shtctl->top); /* 今のマウスと同じ高さになるように指定： マウスはこの上になる */
454  reg[7] = (int) sht;
455  } else if (edx == 6) {
456  sht = (struct SHEET *) (ebx & 0xfffffffe);
457  putfonts8_asc(sht->buf, sht->bxsize, esi, edi, eax, (char *) ebp + ds_base);
458  if ((ebx & 1) == 0) {
459  sheet_refresh(sht, esi, edi, esi + ecx * 8, edi + 16);
460  }
461  } else if (edx == 7) {
462  sht = (struct SHEET *) (ebx & 0xfffffffe);
463  boxfill8(sht->buf, sht->bxsize, ebp, eax, ecx, esi, edi);
464  if ((ebx & 1) == 0) {
465  sheet_refresh(sht, eax, ecx, esi + 1, edi + 1);
466  }
467  } else if (edx == 8) {
468  memman_init((struct MEMMAN *) (ebx + ds_base));
469  ecx &= 0xfffffff0; /* 16バイト単位に */
470  memman_free((struct MEMMAN *) (ebx + ds_base), eax, ecx);
471  } else if (edx == 9) {
472  ecx = (ecx + 0x0f) & 0xfffffff0; /* 16バイト単位に切り上げ */
473  reg[7] = memman_alloc((struct MEMMAN *) (ebx + ds_base), ecx);
474  } else if (edx == 10) {
475  ecx = (ecx + 0x0f) & 0xfffffff0; /* 16バイト単位に切り上げ */
476  memman_free((struct MEMMAN *) (ebx + ds_base), eax, ecx);
477  } else if (edx == 11) {
478  sht = (struct SHEET *) (ebx & 0xfffffffe);
479  sht->buf[sht->bxsize * edi + esi] = eax;
480  if ((ebx & 1) == 0) {
481  sheet_refresh(sht, esi, edi, esi + 1, edi + 1);
482  }
483  } else if (edx == 12) {
484  sht = (struct SHEET *) ebx;
485  sheet_refresh(sht, eax, ecx, esi, edi);
486  } else if (edx == 13) {
487  sht = (struct SHEET *) (ebx & 0xfffffffe);
488  hrb_api_linewin(sht, eax, ecx, esi, edi, ebp);
489  if ((ebx & 1) == 0) {
490  if (eax > esi) {
491  i = eax;
492  eax = esi;
493  esi = i;
494  }
495  if (ecx > edi) {
496  i = ecx;
497  ecx = edi;
498  edi = i;
499  }
500  sheet_refresh(sht, eax, ecx, esi + 1, edi + 1);
501  }
502  } else if (edx == 14) {
503  sheet_free((struct SHEET *) ebx);
504  } else if (edx == 15) {
505  for (;;) {
506  io_cli();
507  if (fifo32_status(&task->fifo) == 0) {
508  if (eax != 0) {
509  task_sleep(task); /* FIFOが空なので寝て待つ */
510  } else {
511  io_sti();
512  reg[7] = -1;
513  return 0;
514  }
515  }
516  i = fifo32_get(&task->fifo);
517  io_sti();
518  if (i <= 1 && cons->sht != 0) { /* カーソル用タイマ */
519  /* アプリ実行中はカーソルが出ないので、いつも次は表示用の1を注文しておく */
520  timer_init(cons->timer, &task->fifo, 1); /* 次は1を */
521  timer_settime(cons->timer, 50);
522  }
523  if (i == 2) { /* カーソルON */
524  cons->cur_c = COL8_FFFFFF;
525  }
526  if (i == 3) { /* カーソルOFF */
527  cons->cur_c = -1;
528  }
529  if (i == 4) { /* コンソールだけを閉じる */
530  timer_cancel(cons->timer);
531  io_cli();
532  fifo32_put(sys_fifo, cons->sht - shtctl->sheets0 + 2024); /* 2024〜2279 */
533  cons->sht = 0;
534  io_sti();
535  }
536  if (i >= 256) { /* キーボードデータ（タスクA経由）など */
537  reg[7] = i - 256;
538  return 0;
539  }
540  }
541  } else if (edx == 16) {
542  reg[7] = (int) timer_alloc();
543  ((struct TIMER *) reg[7])->flags2 = 1; /* 自動キャンセル有効 */
544  } else if (edx == 17) {
