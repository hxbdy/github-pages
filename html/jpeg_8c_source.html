148 unsigned short get_bits(JPEG *jpeg, int bit)
149 {
150  unsigned char c, c2;
151  unsigned short ret;
152  unsigned long buff;
153  int remain;
154 
155  buff = jpeg->bit_buff;
156  remain = jpeg->bit_remain;
157 
158  while (remain <= 16) {
159  if (jpeg->fp >= jpeg->fp1) {
160  ret = 0;
161  goto fin;
162  }
163  c = *jpeg->fp++;
164  if (c == 0xff) { // マーカエラーを防ぐため、FF -> FF 00 にエスケープされてる
165  if (jpeg->fp >= jpeg->fp1) {
166  ret = 0;
167  goto fin;
168  }
169  jpeg->fp++; /* 00をskip */
170  }
171  buff = (buff << 8) | c;
172  remain += 8;
173  }
174  ret = (buff >> (remain - bit)) & ((1 << bit) - 1);
175  remain -= bit;
176 
177  jpeg->bit_remain = remain;
178  jpeg->bit_buff = buff;
179 fin:
180  return ret;
181 }

262 // define huffman table
263 int jpeg_dht(JPEG *jpeg)
264 {
265  unsigned tc, th;
266  unsigned code = 0;
267  unsigned char val;
268  int i, j, k, num, Li[17];
269  int len, max_val;
270  HUFF *table;
271 
272  if (jpeg->fp + 2 > jpeg->fp1)
273  goto err;
274  len = (jpeg->fp[0] << 8 | jpeg->fp[1]) - 2;
275  jpeg->fp += 2;
276 
277  while (len > 0) {
278  if (jpeg->fp + 17 > jpeg->fp1)
279  goto err;
280  val = jpeg->fp[0];
281 
282  tc = (val >> 4) & 0x0f; // テーブルクラス(DC/AC成分セレクタ)
283  th = val & 0x0f; // テーブルヘッダ(何枚目のプレーンか)
284  table = &(jpeg->huff[tc][th]);
285 
286  num = 0;
287  k = 0;
288  for (i = 1; i <= 16; i++) {
289  Li[i] = jpeg->fp[i];
290  num += Li[i];
291  for (j = 0; j < Li[i]; j++)
292  table->size[k++] = i;
293  }
294  table->elem = num;
295  jpeg->fp += 17;
296 
297  k=0;
298  code=0;
299  i = table->size[0];
300  while (k < num) {
301  while (table->size[k] == i)
302  table->code[k++] = code++;
303  if (k >= num)
304  break;
305  do {
306  code <<= 1;
307  i++;
308  } while (table->size[k] != i);
309  }
310 
311  if (jpeg->fp + num > jpeg->fp1)
312  goto err;
313  for (k = 0; k < num; k++)
314  table->value[k] = jpeg->fp[k];
315  jpeg->fp += num;
316 
317  len -= 18 + num;
318  }
319  return 0;
320 err:
321  jpeg->fp = jpeg->fp1;
322  return 1;
323 }

460 void jpeg_idct_init(int base_img[64][64])
461 {
462  int u, v, m, n;
463  int tmpm[8], tmpn[8];
464  int cost[32];
465  cost[ 0] = 32768; cost[ 1] = 32138; cost[ 2] = 30274; cost[ 3] = 27246; cost[ 4] = 23170; cost[ 5] = 18205; cost[ 6] = 12540; cost[ 7] = 6393;
466  cost[ 8] = 0; cost[ 9] = -6393; cost[10] = -12540; cost[11] = -18205; cost[12] = -23170; cost[13] = -27246; cost[14] = -30274; cost[15] = -32138;
467  for (u = 0; u < 16; u++)
468  cost[u + 16] = - cost[u];
469 
470  for (u = 0; u < 8; u++) {
471  {
472  int i=u, d=u*2;
473  if (d == 0)
474  i = 4;
475  for (m = 0; m < 8; m++){
476  tmpm[m] = cost[i]; // 横のCos波形
477  i=(i+d)&31;
478  }
479  }
480  for (v = 0; v < 8; v++) {
481  {
482  int i=v,d=v*2;
483  if (d == 0)
484  i=4;
485  for (n = 0; n < 8; n++){
486  tmpn[n] = cost[i]; // 縦のCos波形
487  i=(i+d)&31;
488  }
489  }
490  // 掛け算して基底画像に
491  for (m = 0; m < 8; m++) {
492  for (n = 0; n < 8; n++) {
493  base_img[u * 8 + v][m * 8 + n] = (tmpm[m] * tmpn[n])>>15;
494  }
495  }
496  }
497  }
498  return;

539 {

545  if(size < 0)
546  return 0;
547  val = jpeg_get_value(jpeg,size);
548  jpeg->mcu_preDC[scan] += val;
549  block[0] = jpeg->mcu_preDC[scan] * pQt[0];
550 
551  //AC復号
552  index = 1;
553  while(index<64)
554  {

556  if(size < 0)
557  break;
558  // EOB
559  if(size == 0)
560  break;
561 
562  // RLE
563  run = (size>>4)&0xF;
564  size = size & 0x0F;
565 
566  val = jpeg_get_value(jpeg,size);
567  if(val>=0x10000) {
568  //マーカ発見
569  return val;
570  }
571 
572  // ZRL
573  while (run-- > 0)
574  block[ zigzag_table[index++] ] = 0;
575 
576  block[ zigzag_table[index] ] = val * pQt[index];
577  index++;
578  }
579  while(index<64)
580  block[zigzag_table[index++]]=0;
581  return 0;
582 }

642 int jpeg_decode_yuv(JPEG *jpeg, int h, int v, unsigned char *rgb, int b_type)
643 {
644  int x0, y0, x, y, x1, y1;
645  int *py;
646  int Y12, V;
647  int mw, mh, w;
648  int i;
649 
650  mw = jpeg->mcu_width;
651  mh = jpeg->mcu_height;
652 
653  x0 = h * jpeg->max_h * 8;
654  y0 = v * jpeg->max_v * 8;
655 
656  x1 = jpeg->width - x0;
657  if (x1 > mw)
658  x1 = mw;
659  y1 = jpeg->height - y0;
660  if (y1 > mh)
661  y1 = mh;
662 
663  py = jpeg->mcu_buf;
664  rgb += (y0 * jpeg->width_buf + x0) * (b_type & 0x7f);
665  w = (jpeg->width_buf - x1) * (b_type & 0x7f);
666 
667  for (y = 0; y < y1; y++) {
668  for (x = 0; x < x1; x++) {
669  int b, g, r;
670  Y12 = py[0] << 12;
671  // U = py[1024]; /* pu */
672  V = py[2048]; /* pv */
673 
674  /* blue */
675  b = 128 + ((Y12 - V * 4 + py[1024] /* pu */ * 0x1C59) >> 12);
676  if (b & 0xffffff00)
677  b = (~b) >> 24;
678 
679  /* green */
680  g = 128 + ((Y12 - V * 0x0B6C) >> 12);
681  if (g & 0xffffff00)
682  g = (~g) >> 24;
683 
684  /* red */
685  r = 128 + ((Y12 + V * 0x166E) >> 12);
686  if (r & 0xffffff00)
687  r = (~r) >> 24;
